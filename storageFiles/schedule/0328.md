 void Die()
 {
     for (int x = 0; x < dropOnDeath.Length; x++)
     {
         // 이렇게 하면 내가 죽었을때 내가 가지고 있는 dropOnDeath 가지고 아이템을 만들어 냅니다
         Instantiate(dropOnDeath[x].dropPrefab, transform.position + Vector3.up * 2, Quaternion.identity);
     }

     Destroy(gameObject);
 }
